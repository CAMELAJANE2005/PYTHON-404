def get_html_content(url):
    response = requests.get(url)
    if response.status_code == 200:
        return response.content
    else:
        return None

def parse_html_content(html):
    soup = BeautifulSoup(html, 'html.parser')
    return soup

def extract_article_text(soup):
    article_text = {}
    current_heading = None

    for element in soup.find_all(['h2', 'p']):
        if element.name == 'h2':
            current_heading = element.text.strip()
            article_text[current_heading] = []
        elif element.name == 'p' and current_heading is not None:
            article_text[current_heading].append(element.text.strip())

    return article_text

def collect_redirect_links(soup):
    redirect_links = []

    for link in soup.find_all('a', href=True):
        href = link['href']
        if href.startswith('/wiki/'):
            redirect_links.append(href)

    return redirect_links

def process_wikipedia_page(url):
    html = get_html_content(url)
    if html:
        soup = parse_html_content(html)
        title = extract_article_title(soup)
        article_text = extract_article_text(soup)
        redirect_links = collect_redirect_links(soup)

        return {
            'title': title,
            'article_text': article_text,
            'redirect_links': redirect_links
        }
    else:
        return None
